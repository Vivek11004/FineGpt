# ====== Stage 1: Base ======
FROM python:3.10-slim

# Set working directory
WORKDIR /app

# Copy backend and frontend
COPY backend ./backend
COPY frontend ./frontend

# Install dependencies
RUN pip install --no-cache-dir -r backend/requirements.txt -r frontend/requirements.txt

# Expose both FastAPI and Streamlit ports
EXPOSE 8000
EXPOSE 8501

# ====== Stage 2: Run both apps ======
CMD bash -c "uvicorn backend.app.main:app --host 0.0.0.0 --port 8000 & \
              streamlit run frontend/app.py --server.port 8501 --server.address 0.0.0.0"
